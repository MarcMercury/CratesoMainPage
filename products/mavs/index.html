For Developers

Full API access to build custom voice experiences

mavs-example.py 
from mavs import MAVSDevice, Agent

# Initialize MAVS device
device = MAVSDevice()

# Register your agents
device.register_agent(Agent(
    name="assistant",
    provider="openai",
    model="gpt-4o"
))

# Handle voice input
@device.on_voice
async def handle_voice(audio, agent):
    response = await agent.process(audio)
    return response

device.start()
